f [2, 4, 0, 3]
> f (x=2:xs=[4, 0, 3]]) = (2*(length [4, 0, 3]])):(f [4, 0, 3])
> (2*3):f[4, 0, 3]
> 6 : f [4, 0, 3]
-- f (x=4:xs=[0, 3]) = (4*(length [0,3])):(f [0,3])
> 6 : (4*(length [0,3])):(f [0,3])
> 6 : (4*2) : f [0, 3]
-- f (x=0:xs=[3]) = (0*(length [3])):(f [3])
-- f [3] = [] (f [_] = [])
> 6 : 8 : 0 : []
> [6, 8, 0]
